    <script>
        // --- Configuration ---
        const API_KEY = '7c8e74d94f46b889acfa7bf61e6fb8ae';
        const BASE_URL = 'https://api.themoviedb.org/3';
        const IMAGE_BASE = 'https://image.tmdb.org/t/p/w500';
        const BACKDROP_BASE = 'https://image.tmdb.org/t/p/original';

        // --- State ---

        // --- TV Remote Navigation (Spatial) ---
        function setupKeyboardNavigation() {
            document.addEventListener('keydown', (e) => {
                const key = e.key;
                
                // Allow default for inputs
                if (document.activeElement.tagName === 'INPUT' && key !== 'Escape' && key !== 'Enter') return;
                
                // Back button on Android TV usually maps to Escape or Backspace
                if (key === 'Escape' || key === 'Backspace') {
                    if (!document.getElementById('trailer-overlay').classList.contains('hidden')) {
                        closeTrailer();
                    } else if (!document.getElementById('detail-modal').classList.contains('hidden')) {
                        closeDetailModal();
                    } else if (document.getElementById('sidebar').classList.contains('open')) {
                        toggleSidebar();
                    }
                    return;
                }

                // D-Pad Navigation logic could be complex, but let's try a simple approach first:
                // Let browser handle native tab navigation, but ensure focus is visible (CSS).
                // We add a listener to ensure arrow keys work if focus is lost.
                if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(key)) {
                    if (document.activeElement === document.body) {
                        // Focus first interactive element
                        const first = document.querySelector('a, button, input');
                        if (first) first.focus();
                    }
                }
            });
        }

        // --- Routing ---
        window.addEventListener('hashchange', handleRouting);

        function handleRouting() {
            const hash = window.location.hash.substring(1) || 'home';
            
            // Update Sidebar
            document.querySelectorAll('.sidebar-link').forEach(l => l.classList.remove('active'));
            const activeLink = document.querySelector(`.sidebar-link[href="#${hash}"]`);
            if (activeLink) activeLink.classList.add('active');

            // Reset State
            page = 1;
            hasMore = true;
            isLoading = false;
            genreId = null;
            document.getElementById('content-grid').innerHTML = '';
            document.getElementById('genres-container').classList.add('hidden');

            switch(hash) {
                case 'home':
                    document.getElementById('page-title').innerText = 'ğŸ”¥ ×—× ×¢×›×©×™×•';
                    document.getElementById('genres-container').classList.remove('hidden');
                    fetchTrending();
                    break;
                case 'movies':
                    document.getElementById('page-title').innerText = 'ğŸ¬ ×¡×¨×˜×™× ×¤×•×¤×•×œ×¨×™×™×';
                    document.getElementById('genres-container').classList.remove('hidden');
                    fetchDiscover('movie');
                    break;
                case 'series':
                    document.getElementById('page-title').innerText = 'ğŸ“º ×¡×“×¨×•×ª ××•×‘×™×œ×•×ª';
                    document.getElementById('genres-container').classList.remove('hidden');
                    fetchDiscover('tv');
                    break;
                case 'cinemas':

                    break;
                case 'new':
                    document.getElementById('page-title').innerText = 'ğŸ†• ×—×“×© ×¢×œ ×”××¡×š';
                    fetchUpcoming();
                    break;

                    renderSettings();
                    break;
                default:
                    fetchTrending();
            }
            currentView = hash;
        }

        // --- API Calls ---
        async function fetchAPI(endpoint, params = {}) {
            const url = new URL(`${BASE_URL}${endpoint}`);
            url.searchParams.append('api_key', API_KEY);
            url.searchParams.append('language', LANGUAGE);
            Object.keys(params).forEach(key => url.searchParams.append(key, params[key]));
            
            try {
                const res = await fetch(url);
                return await res.json();
            } catch (e) {
                console.error(e);
                return null;
            }
        }

        async function fetchGenres() {
            const [m, t] = await Promise.all([
                fetchAPI('/genre/movie/list'),
                fetchAPI('/genre/tv/list')
            ]);
            if (m) m.genres.forEach(g => genres[g.id] = g.name);
            if (t) t.genres.forEach(g => genres[g.id] = g.name);
            renderGenres();
        }

        function renderGenres() {
            const container = document.getElementById('genres-container');
            container.innerHTML = `<button onclick="filterGenre(null)" class="px-4 py-1 rounded-full glass-btn hover:bg-red-600/50 bg-red-600" tabindex="0">×”×›×œ</button>`;
            // Top genres
            const topGenres = [28, 35, 18, 10751, 16, 27, 878];
            topGenres.forEach(id => {
                if (genres[id]) {
                    container.innerHTML += `<button onclick="filterGenre(${id})" class="px-4 py-1 rounded-full glass-btn hover:bg-white/20" tabindex="0">${genres[id]}</button>`;
                }
            });
        }

        window.filterGenre = (id) => {
            genreId = id;
            page = 1;
            document.getElementById('content-grid').innerHTML = '';
            
            // Update UI
            const btns = document.querySelectorAll('#genres-container button');
            btns.forEach(b => {
                b.classList.remove('bg-red-600', 'bg-red-600/50');
                if ((id === null && b.innerText === '×”×›×œ') || (id && b.innerText === genres[id])) {
                    b.classList.add('bg-red-600');
                }
            });

            if (currentView === 'home') fetchTrending();
            else if (currentView === 'movies') fetchDiscover('movie');
            else if (currentView === 'series') fetchDiscover('tv');
        };

        async function fetchTrending() {
            if (isLoading || !hasMore) return;
            isLoading = true;
            document.getElementById('loading').classList.remove('hidden');
            
            let data;
            if (genreId) {
                data = await fetchAPI('/discover/movie', { with_genres: genreId, page, sort_by: 'popularity.desc' });
            } else {
                data = await fetchAPI('/trending/all/week', { page });
            }
            
            processResults(data);
        }

        async function fetchDiscover(type) {
            if (isLoading || !hasMore) return;
            isLoading = true;
            document.getElementById('loading').classList.remove('hidden');
            
            const params = { page, sort_by: 'popularity.desc' };
            if (genreId) params.with_genres = genreId;
            
            const data = await fetchAPI(`/discover/${type}`, params);
            // Inject media_type manually since discover endpoint doesn't always return it
            if (data && data.results) data.results.forEach(i => i.media_type = type);
            processResults(data);
        }

        async function fetchNowPlaying() {
            if (isLoading || !hasMore) return;
            isLoading = true;
            const data = await fetchAPI('/movie/now_playing', { page, region: 'IL' });
            if (data && data.results) data.results.forEach(i => i.media_type = 'movie');
            processResults(data);
        }
        
        async function fetchUpcoming() {
             if (isLoading || !hasMore) return;
            isLoading = true;
            const data = await fetchAPI('/movie/upcoming', { page, region: 'IL' });
            if (data && data.results) data.results.forEach(i => i.media_type = 'movie');
            processResults(data);
        }

        function processResults(data) {
            document.getElementById('loading').classList.add('hidden');
            isLoading = false;
            
            if (!data || !data.results || data.results.length === 0) {
                hasMore = false;
                return;
            }

            const grid = document.getElementById('content-grid');
            data.results.forEach(item => {
                if (!item.poster_path) return;
                
                const title = item.title || item.name;
                const year = (item.release_date || item.first_air_date || '').split('-')[0];
                const type = item.media_type === 'movie' ? '×¡×¨×˜' : '×¡×“×¨×”';
                
                const el = document.createElement('div');
                el.className = 'card cursor-pointer group focus:ring-4 focus:ring-red-600 focus:outline-none';
                el.tabIndex = 0; // Make focusable
                el.innerHTML = `
                    <img src="${IMAGE_BASE}${item.poster_path}" class="card-poster" alt="${title}">
                    <div class="card-overlay opacity-0 group-hover:opacity-100 group-focus:opacity-100 transition-opacity">
                        <h3 class="card-title">${title}</h3>
                        <div class="flex justify-between items-center">
                            <span class="text-sm opacity-80">${year} â€¢ ${type}</span>
                            <span class="card-rating">â˜… ${item.vote_average.toFixed(1)}</span>
                        </div>
                    </div>
                `;
                
                // Click & Enter support
                const open = () => openModal(item.id, item.media_type || (item.title ? 'movie' : 'tv'));
                el.addEventListener('click', open);
                el.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter') open();
                });

                grid.appendChild(el);
            });
            page++;
        }

        // --- Search ---
        function setupSearch() {
            const input = document.getElementById('search-input');
            let timeout;
            input.addEventListener('input', (e) => {
                clearTimeout(timeout);
                timeout = setTimeout(() => {
                    const query = e.target.value;
                    if (query) {
                        document.getElementById('page-title').innerText = `×ª×•×¦××•×ª ×—×™×¤×•×©: ${query}`;
                        document.getElementById('content-grid').innerHTML = '';
                        document.getElementById('genres-container').classList.add('hidden');
                        page = 1;
                        hasMore = false; // Disable infinite scroll for search simply
                        performSearch(query);
                    } else {
                        handleRouting(); // Reset
                    }
                }, 500);
            });
        }

        async function performSearch(query) {
            document.getElementById('loading').classList.remove('hidden');
            const data = await fetchAPI('/search/multi', { query });
            document.getElementById('loading').classList.add('hidden');
            
            if (data && data.results) {
                // Filter only movie/tv
                const filtered = { results: data.results.filter(i => i.media_type === 'movie' || i.media_type === 'tv') };
                processResults(filtered);
            }
        }

        // --- Detail Modal ---
        async function openModal(id, type) {
            const modal = document.getElementById('detail-modal');
            const backdrop = document.getElementById('modal-backdrop');
            
            // Show basic info immediately while fetching
            modal.classList.remove('hidden');
            document.body.style.overflow = 'hidden';

            const details = await fetchAPI(`/${type}/${id}`, { append_to_response: 'videos,credits,external_ids' });
